# Macro Tab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Macro tab showing 5 market-level signals (VIX, S5FI, Fear & Greed, FedWatch, SPY vs ATH) with deterministic labels, explanations, composite posture, and an `ai_context` field for future AI synthesis.

**Architecture:** Single API endpoint (`api/macro.js`) fetches all signals in parallel, computes labels/scores/posture server-side, returns combined response. React hook (`useMacro`) polls this endpoint. `MacroTab.jsx` renders a posture header + 5 signal cards in a responsive grid. Daily cron writes macro data to `macro_snapshots` Supabase table.

**Tech Stack:** Vercel serverless functions, Yahoo Finance, CNN dataviz, Finviz HTML scraping, rateprobability.com, Supabase, React with inline styles using `theme.js` tokens.

```js
async function fetchVixAndSpy() {
  // Fetch VIX and SPY in a single parallel call — both from Yahoo Finance
  const [vixRes, spyRes] = await Promise.all([
    fetch("https://query1.finance.yahoo.com/v8/finance/chart/%5EVIX?interval=1d&range=1d", {
      headers: { "User-Agent": UA, Accept: "application/json" },
    }),
    fetch("https://query1.finance.yahoo.com/v8/finance/chart/SPY?interval=1d&range=1d", {
      headers: { "User-Agent": UA, Accept: "application/json" },
    }),
  ]);

  const vixData = await vixRes.json();
  const spyData = await spyRes.json();

  const vixMeta = vixData?.chart?.result?.[0]?.meta;
  const spyMeta = spyData?.chart?.result?.[0]?.meta;

  return {
    vix: vixMeta?.regularMarketPrice ?? null,
    vixPrevClose: vixMeta?.chartPreviousClose ?? null,
    spyPrice: spyMeta?.regularMarketPrice ?? null,
    spy52wHigh: spyMeta?.fiftyTwoWeekHigh ?? null,
  };
}

async function fetchFearGreed() {
  const today = new Date().toISOString().split("T")[0];
  const res = await fetch(
    `https://production.dataviz.cnn.io/index/fearandgreed/graphdata/${today}`,
    {
      headers: {
        "User-Agent": UA,
        Referer: "https://www.cnn.com/markets/fear-and-greed",
      },
    }
  );
  if (!res.ok) return null;
  const data = await res.json();
  const fg = data?.fear_and_greed;
  if (!fg) return null;
  return {
    score: fg.score,
    rating: fg.rating,
    previousClose: fg.previous_close,
    previous1w: fg.previous_1_week,
    previous1m: fg.previous_1_month,
    previous1y: fg.previous_1_year,
  };
}

async function fetchS5fi() {
  const [totalRes, aboveRes] = await Promise.all([
    fetch("https://finviz.com/screener.ashx?v=111&f=idx_sp500&ft=4", {
      headers: { "User-Agent": UA },
    }),
    fetch("https://finviz.com/screener.ashx?v=111&f=idx_sp500,ta_sma50_pa&ft=4", {
      headers: { "User-Agent": UA },
    }),
  ]);
  const totalHtml = await totalRes.text();
  const aboveHtml = await aboveRes.text();

  const totalMatch = totalHtml.match(/#1\s*\/\s*(\d+)/);
  const total = totalMatch ? parseInt(totalMatch[1], 10) : null;

  let above = 0;
  if (aboveHtml.includes("No results")) {
    above = 0;
  } else {
    const aboveMatch = aboveHtml.match(/#1\s*\/\s*(\d+)/);
    above = aboveMatch ? parseInt(aboveMatch[1], 10) : null;
  }

  if (total == null || above == null || total === 0) return null;
  return Math.round((above / total) * 1000) / 10; // one decimal, e.g. 52.1
}

async function fetchFedWatch() {
  const res = await fetch("https://rateprobability.com/api/latest", {
    headers: { "User-Agent": UA },
  });
  if (!res.ok) return null;
  const data = await res.json();
  const today = data?.today;
  if (!today?.rows?.length) return null;

  const rows = today.rows;
  const cutoff = new Date();
  cutoff.setFullYear(cutoff.getFullYear() + 1);

  const next12mRows = rows.filter((r) => new Date(r.meeting_iso) <= cutoff);
  const last12m = next12mRows[next12mRows.length - 1];
  const cutsPricedIn = last12m?.num_moves_is_cut ? last12m.num_moves : last12m?.num_moves ? -last12m.num_moves : 0;

  const decRow = rows.find((r) => r.meeting_iso.startsWith("2026-12"));
  const endOfYearImplied = decRow?.implied_rate_post_meeting ?? null;

  const nextMeeting = rows[0];

  // Weekly change
  const ago1w = data?.ago_1w;
  let weeklyChangeBps = null;
  if (ago1w?.rows?.length && endOfYearImplied != null) {
    const ago1wDec = ago1w.rows.find((r) => r.meeting_iso.startsWith("2026-12"));
    if (ago1wDec?.implied_rate_post_meeting != null) {
      weeklyChangeBps = Math.round((endOfYearImplied - ago1wDec.implied_rate_post_meeting) * 100);
    }
  }

  // Curve direction
  const firstImplied = rows[0]?.implied_rate_post_meeting;
  const lastImplied = rows[rows.length - 1]?.implied_rate_post_meeting;
  const curveDirection = lastImplied < firstImplied ? "declining" : lastImplied > firstImplied ? "rising" : "flat";

  return {
    cutsPricedIn: Math.round(cutsPricedIn * 100) / 100,
    endOfYearImplied: endOfYearImplied != null ? Math.round(endOfYearImplied * 1000) / 1000 : null,
    currentRate: today.midpoint,
    nextMeetingDate: nextMeeting?.meeting_iso ?? null,
    weeklyChangeBps,
    curveDirection,
  };
}

// ── Composite posture ────────────────────────────────────────────────────────

function computePosture(signals) {
  const scores = [
    signals.vix.score,
    signals.s5fi.score,
    signals.fearGreed.score,
    signals.fedWatch.score,
    signals.spyVsAth.score,
  ];
  const avg = scores.reduce((a, b) => a + b, 0) / scores.length;

  let posture, description, deploymentGuidance;
  if (avg >= 4.2) {
    posture = "BULLISH";
    description = "All signals aligned positively. Strong conditions for premium selling.";
    deploymentGuidance = "Deploy aggressively. Chase stocks higher with tighter strikes. VIX permitting, push toward lower cash floor.";
  } else if (avg >= 3.4) {
    posture = "CONSTRUCTIVE";
    description = "Most signals positive with minor headwinds. Good environment for the wheel strategy.";
    deploymentGuidance = "Normal deployment. Follow Ryan signals. Stay within VIX cash bands. Let winning positions run.";
  } else if (avg >= 2.6) {
    posture = "NEUTRAL";
    description = "Mixed signals — some positive, some negative. Proceed with caution.";
    deploymentGuidance = "Selective deployment only. Higher-quality names, wider strikes. Hold 20-25% cash. Wait for clearer signals.";
  } else if (avg >= 1.8) {
    posture = "DEFENSIVE";
    description = "Majority of signals negative. Elevated risk environment.";
    deploymentGuidance = "Minimal new deployment. Focus on managing existing positions. 25-35% cash. Consider bear call spreads as hedge.";
  } else {
    posture = "BEARISH";
    description = "Signals aligned negatively. Capital preservation mode.";
    deploymentGuidance = "No new CSPs. Manage assignments defensively. Maximum cash position. Watch for S5FI single-digit readings as a bottom signal.";
  }

  return { posture, avg: Math.round(avg * 10) / 10, scores, description, deploymentGuidance };
}

export default async function handler(req, res) {
  if (req.method !== "GET") {
    return res.status(405).json({ ok: false, error: "Method not allowed" });
  }

  try {
    const [yahooData, fearGreedData, s5fiPct, fedWatchData] = await Promise.allSettled([
      fetchVixAndSpy(),
      fetchFearGreed(),
      fetchS5fi(),
      fetchFedWatch(),
    ]);

    // Extract values with fallbacks
    const yahoo = yahooData.status === "fulfilled" ? yahooData.value : {};
    const fg    = fearGreedData.status === "fulfilled" ? fearGreedData.value : null;
    const s5fi  = s5fiPct.status === "fulfilled" ? s5fiPct.value : null;
    const fw    = fedWatchData.status === "fulfilled" ? fedWatchData.value : null;

    const vixVal = yahoo.vix ?? null;
    const vixChange = vixVal != null && yahoo.vixPrevClose != null
      ? Math.round((vixVal - yahoo.vixPrevClose) * 100) / 100
      : null;

    const spyPctFromHigh = yahoo.spyPrice != null && yahoo.spy52wHigh != null && yahoo.spy52wHigh > 0
      ? (yahoo.spyPrice - yahoo.spy52wHigh) / yahoo.spy52wHigh
      : null;

    // Build labeled signals
    const signals = {
      vix: {
        value: vixVal,
        change: vixChange,
        ...(vixVal != null ? labelVix(vixVal) : { score: 3, label: "Unavailable", color: "amber", explanation: "VIX data unavailable." }),
      },
      s5fi: {
        value: s5fi,
        ...(s5fi != null ? labelS5fi(s5fi) : { score: 3, label: "Unavailable", color: "amber", explanation: "S5FI data unavailable." }),
      },
      fearGreed: {
        value: fg?.score ?? null,
        prev1w: fg?.previous1w ?? null,
        prev1m: fg?.previous1m ?? null,
        prev1y: fg?.previous1y ?? null,
        previousClose: fg?.previousClose ?? null,
        ...(fg?.score != null ? labelFearGreed(fg.score) : { score: 3, label: "Unavailable", color: "amber", explanation: "Fear & Greed data unavailable." }),
      },
      fedWatch: {
        cutsPricedIn: fw?.cutsPricedIn ?? null,
        endOfYearImplied: fw?.endOfYearImplied ?? null,
        currentRate: fw?.currentRate ?? null,
        nextMeetingDate: fw?.nextMeetingDate ?? null,
        weeklyChangeBps: fw?.weeklyChangeBps ?? null,
        curveDirection: fw?.curveDirection ?? null,
        ...(fw != null ? labelFedWatch(fw) : { score: 2, label: "Unavailable", color: "amber", direction: null, explanation: "FedWatch data unavailable." }),
      },
      spyVsAth: {
        value: yahoo.spyPrice ?? null,
        high: yahoo.spy52wHigh ?? null,
        pctFromHigh: spyPctFromHigh,
        ...(spyPctFromHigh != null ? labelSpyVsAth(spyPctFromHigh) : { score: 3, label: "Unavailable", color: "amber", explanation: "SPY data unavailable." }),
      },
    };

    const posture = computePosture(signals);
    const aiContext = buildAIContext(signals, posture);

    res.setHeader("Cache-Control", "s-maxage=1800, stale-while-revalidate=300");
    res.status(200).json({
      ok: true,
      as_of: new Date().toISOString(),
      posture,
      signals,
      ai_context: aiContext,
    });
  } catch (err) {
    console.error("[api/macro] Error:", err);
    res.status(500).json({ ok: false, error: err.message });
  }
}
```

```js
import { useState, useEffect, useCallback } from "react";

export function useMacro() {
  const [data, setData]       = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError]     = useState(null);

  const fetchMacro = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const res = await fetch("/api/macro");
      const json = await res.json();
      if (!json.ok) {
        setError(json.error || "Failed to load macro data");
      } else {
        setData(json);
      }
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { fetchMacro(); }, [fetchMacro]);

  return { data, loading, error, refresh: fetchMacro };
}
```

```jsx
import { useMacro } from "../hooks/useMacro";
import { useWindowWidth } from "../hooks/useWindowWidth";

// ── Posture colors (intentional hardcoded hex — semantic data like TYPE_COLORS) ──
const POSTURE_COLORS = {
  BULLISH:      { text: "#3fb950", bg: "#0d1f0d" },
  CONSTRUCTIVE: { text: "#3fb950", bg: "#0d1f0d" },
  NEUTRAL:      { text: "#e3b341", bg: "#1f1a0d" },
  DEFENSIVE:    { text: "#e3b341", bg: "#1f1a0d" },
  BEARISH:      { text: "#f85149", bg: "#1f0d0d" },
};

const SIGNAL_COLORS = {
  green: theme.green,
  amber: theme.amber,
  red:   theme.red,
};

// ── ScoreDots ────────────────────────────────────────────────────────────────

function ScoreDots({ score, max = 5, color }) {
  return (
    <div style={{ display: "flex", gap: 3 }}>
      {Array.from({ length: max }, (_, i) => (
        <div key={i} style={{
          width: 8, height: 8, borderRadius: "50%",
          background: i < score ? (SIGNAL_COLORS[color] || theme.text.muted) : theme.border.default,
        }} />
      ))}
    </div>
  );
}

// ── SignalCard ────────────────────────────────────────────────────────────────

function SignalCard({ name, value, label, color, score, explanation, direction, children }) {
  const [expanded, setExpanded] = useState(false);

  return (
    <div style={{
      background: theme.bg.surface,
      border: `1px solid ${theme.border.default}`,
      borderRadius: theme.radius.md,
      padding: theme.space[4],
      display: "flex",
      flexDirection: "column",
      gap: theme.space[2],
    }}>
      {/* Header row */}
      <div style={{ fontSize: theme.size.xs, color: theme.text.subtle, textTransform: "uppercase", letterSpacing: "0.5px" }}>
        {name}
      </div>

      {/* Value + Label row */}
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "baseline" }}>
        <span style={{ fontSize: theme.size.xl, fontWeight: 600, color: theme.text.primary }}>
          {value ?? "—"}
        </span>
        <span style={{
          fontSize: theme.size.sm,
          fontWeight: 500,
          color: SIGNAL_COLORS[color] || theme.text.muted,
        }}>
          {label}
        </span>
      </div>

      {/* Direction / trend (optional) */}
      {direction && (
        <div style={{ fontSize: theme.size.xs, color: theme.text.muted }}>
          {direction}
        </div>
      )}

      {/* Extra content (e.g. FedWatch details, F&G trend) */}
      {children}

      {/* Score dots */}
      <ScoreDots score={score} color={color} />

      {/* Expandable explanation */}
      <div
        onClick={() => setExpanded(!expanded)}
        style={{ fontSize: theme.size.xs, color: theme.text.subtle, cursor: "pointer", userSelect: "none" }}
      >
        {expanded ? "▴ " : "▾ "}Why this matters
      </div>
      {expanded && (
        <div style={{ fontSize: theme.size.sm, color: theme.text.muted, lineHeight: 1.5 }}>
          {explanation}
        </div>
      )}
    </div>
  );
}

// ── MacroTab ─────────────────────────────────────────────────────────────────

export function MacroTab() {
  const { data, loading, error, refresh } = useMacro();
  const windowWidth = useWindowWidth();
  const isMobile = windowWidth < 768;

  if (loading) {
    return (
      <div style={{ color: theme.text.muted, padding: theme.space[5], textAlign: "center" }}>
        Loading macro signals...
      </div>
    );
  }

  if (error || !data) {
    return (
      <div style={{ color: theme.red, padding: theme.space[5], textAlign: "center" }}>
        Failed to load macro data{error ? `: ${error}` : ""}
        <div style={{ marginTop: theme.space[3] }}>
          <button onClick={refresh} style={{
            background: theme.bg.surface, border: `1px solid ${theme.border.strong}`,
            color: theme.text.secondary, padding: "6px 16px", borderRadius: theme.radius.sm,
            cursor: "pointer", fontFamily: "inherit", fontSize: theme.size.sm,
          }}>
            Retry
          </button>
        </div>
      </div>
    );
  }

  const { posture, signals, as_of } = data;
  const postureColor = POSTURE_COLORS[posture.posture] || POSTURE_COLORS.NEUTRAL;

  // Stale check: > 3 hours old
  const asOfDate = new Date(as_of);
  const isStale = (Date.now() - asOfDate.getTime()) > 3 * 60 * 60 * 1000;
  const asOfStr = asOfDate.toLocaleString("en-US", {
    month: "short", day: "numeric", year: "numeric",
    hour: "numeric", minute: "2-digit",
  });

  const { vix, s5fi, fearGreed, fedWatch, spyVsAth } = signals;

  return (
    <div>
      {/* ── Posture header ── */}
      <div style={{
        background: postureColor.bg,
        border: `1px solid ${theme.border.default}`,
        borderRadius: theme.radius.md,
        padding: theme.space[5],
        marginBottom: theme.space[5],
      }}>
        <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", marginBottom: theme.space[3], flexWrap: "wrap", gap: theme.space[2] }}>
          <div style={{ fontSize: theme.size.xs, color: theme.text.subtle, textTransform: "uppercase", letterSpacing: "1px" }}>
            Macro Outlook
          </div>
          <div style={{ display: "flex", alignItems: "center", gap: theme.space[3], fontSize: theme.size.xs, color: theme.text.subtle }}>
            <span>as of {asOfStr}</span>
            {isStale && <span style={{ color: theme.amber }}>Data may be stale</span>}
            <button onClick={refresh} style={{
              background: "transparent", border: `1px solid ${theme.border.strong}`,
              color: theme.text.muted, padding: "3px 10px", borderRadius: theme.radius.sm,
              cursor: "pointer", fontFamily: "inherit", fontSize: theme.size.xs,
            }}>
              Refresh
            </button>
          </div>
        </div>

        <div style={{ display: "flex", justifyContent: "space-between", alignItems: "baseline", flexWrap: "wrap", gap: theme.space[2] }}>
          <span style={{ fontSize: 28, fontWeight: 700, color: postureColor.text, letterSpacing: "1px" }}>
            {posture.posture}
          </span>
          <span style={{ fontSize: theme.size.md, color: theme.text.muted }}>
            Score: {posture.avg} / 5.0
          </span>
        </div>

        <div style={{ fontSize: theme.size.sm, color: theme.text.secondary, marginTop: theme.space[2], lineHeight: 1.5 }}>
          {posture.deploymentGuidance}
        </div>
      </div>

      {/* ── Signal grid ── */}
      <div style={{
        display: "grid",
        gridTemplateColumns: isMobile ? "1fr 1fr" : "1fr 1fr 1fr",
        gap: theme.space[4],
      }}>
        {/* VIX */}
        <SignalCard
          name="VIX"
          value={vix.value != null ? vix.value.toFixed(2) : null}
          label={vix.label}
          color={vix.color}
          score={vix.score}
          explanation={vix.explanation}
          direction={vix.change != null ? `${vix.change > 0 ? "▲" : "▼"} ${vix.change > 0 ? "+" : ""}${vix.change.toFixed(1)} today` : null}
        />

        {/* S5FI */}
        <SignalCard
          name="S5FI"
          value={s5fi.value != null ? `${s5fi.value.toFixed(0)}%` : null}
          label={s5fi.label}
          color={s5fi.color}
          score={s5fi.score}
          explanation={s5fi.explanation}
          direction="Daily close"
        />

        {/* Fear & Greed */}
        <SignalCard
          name="Fear & Greed"
          value={fearGreed.value != null ? fearGreed.value.toFixed(1) : null}
          label={fearGreed.label}
          color={fearGreed.color}
          score={fearGreed.score}
          explanation={fearGreed.explanation}
        >
          {(fearGreed.prev1w != null || fearGreed.prev1m != null) && (
            <div style={{ fontSize: theme.size.xs, color: theme.text.subtle }}>
              {fearGreed.prev1w != null && <span>1w: {fearGreed.prev1w.toFixed(1)}</span>}
              {fearGreed.prev1w != null && fearGreed.prev1m != null && <span style={{ margin: "0 4px" }}>·</span>}
              {fearGreed.prev1m != null && <span>1m: {fearGreed.prev1m.toFixed(1)}</span>}
            </div>
          )}
        </SignalCard>

        {/* FedWatch */}
        <SignalCard
          name="Rate Expectations"
          value={fedWatch.cutsPricedIn != null ? `${fedWatch.cutsPricedIn.toFixed(1)} cuts` : null}
          label={fedWatch.label}
          color={fedWatch.color}
          score={fedWatch.score}
          explanation={fedWatch.explanation}
          direction={fedWatch.direction ? `↓ ${fedWatch.direction}` : null}
        >
          <div style={{ fontSize: theme.size.xs, color: theme.text.subtle }}>
            {fedWatch.endOfYearImplied != null && (
              <div>Dec 2026 implied: {fedWatch.endOfYearImplied}% (current: {fedWatch.currentRate}%)</div>
            )}
            {fedWatch.nextMeetingDate && (
              <div>Next FOMC: {new Date(fedWatch.nextMeetingDate).toLocaleDateString("en-US", { month: "short", day: "numeric" })}</div>
            )}
          </div>
        </SignalCard>

        {/* SPY vs ATH */}
        <SignalCard
          name="SPY vs ATH"
          value={spyVsAth.pctFromHigh != null ? `${(spyVsAth.pctFromHigh * 100).toFixed(1)}%` : null}
          label={spyVsAth.label}
          color={spyVsAth.color}
          score={spyVsAth.score}
          explanation={spyVsAth.explanation}
        >
          {spyVsAth.value != null && spyVsAth.high != null && (
            <div style={{ fontSize: theme.size.xs, color: theme.text.subtle }}>
              ${spyVsAth.value.toFixed(2)} / ${spyVsAth.high.toFixed(2)} high
            </div>
          )}
        </SignalCard>
      </div>
    </div>
  );
}
```

- [ ] **Step 1: Add macro signal fetch and `macro_snapshots` upsert to the snapshot cron**

After the existing snapshot upsert (line 193 of `api/snapshot.js`), add a macro fetch and write:

```js
// 10. Fetch macro signals and write to macro_snapshots
try {
  const macroRes = await fetch(
    `https://${req.headers.host}/api/macro`,
    { headers: { "User-Agent": "internal-snapshot-cron" } }
  );
  const macroData = await macroRes.json();

  if (macroData.ok) {
    const macroSnapshot = {
      snapshot_date: today,
      vix: macroData.signals.vix.value,
      s5fi_pct: macroData.signals.s5fi.value,
      fear_greed_score: macroData.signals.fearGreed.value,
      fed_cuts_priced_in: macroData.signals.fedWatch.cutsPricedIn,
      spy_pct_from_ath: macroData.signals.spyVsAth.pctFromHigh,
      posture: macroData.posture.posture,
      posture_score: macroData.posture.avg,
      ai_context: macroData.ai_context,
    };

    const { error: macroError } = await supabase
      .from("macro_snapshots")
      .upsert(macroSnapshot, { onConflict: "snapshot_date" });

    if (macroError) {
      console.error("[api/snapshot] Macro snapshot write failed:", macroError);
    }
  }
} catch (macroErr) {
  console.error("[api/snapshot] Macro fetch/write failed:", macroErr.message);
}
```

This is wrapped in try/catch so a macro failure never breaks the existing portfolio snapshot.
